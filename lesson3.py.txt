salary = ['10000','20000','25000','30000']
salary_names = list(zip(names, salary))

with open('salary.txt', 'w', encoding='UTF-8') as file:
    for salary_num in salary_names:
        file.write('{} - {}\n'.format(salary_num[0], salary_num[1]))

    file = file.read()

        salary_list.append(salary_from_file.split(' - '))

        salary_l[0] = str(salary_l[0]).upper()
        salary_2 = int(salary_l[1])
